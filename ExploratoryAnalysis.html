"""Converts all the airport id's into the airport's name"""

idToAirport = {}  # Dictionary holding id to airport

with open('L_AIRPORT_ID.csv_', 'r') as f:
    reader = csv.reader(f)

    for i, row in enumerate(reader):

        # .csv file holds name of airport and city, so split
        if i != 0:
            id = int(row[0])
            airport = row[1].split(': ')[-1]

            # Add to dict
            if id not in idToAirport:
                idToAirport[id] = airport

data['ORIGIN_AIRPORT_ID'] = data['ORIGIN_AIRPORT_ID'].map(idToAirport)

delayRate = []

for i, row in sortedDelayCount_df[:10].iterrows():
    airport = row[0]
    delay = row[1] * 1.0 / flightCounts[airport]
    delayRate.append((airport, delay))
    
delayRate.sort(key=lambda tup: tup[1], reverse=True)
